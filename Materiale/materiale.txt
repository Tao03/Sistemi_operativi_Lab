struct sembuf my_op ;
    my_op . sem_num = 0; /* only one semaphore in array of semaphores */
    my_op . sem_flg = 0; /* no flag : default behavior */
    my_op . sem_op = -1; /* accessing the resource */
    printf("Semaforo settato");
    semop ( id , & my_op , 1) ; /* blocking if others hold resource */
    //printf("ID dell'array di semafori: %d, valore semaforo %d\n",id, semctl(id, 0, GETVAL));
    /**
     *

     * SEZIONE CRITICA
     * 
     * 
     * 
     * 
    */
   my_op . sem_op = 1; /* releasing the resource */
   semop ( id , & my_op , 1) ; /* may un - block others */


   Lista di opperazioni del semaforo, n=sem_op:
    n>0, incrementa il valore del semaforo di n;
    n<0, decrementa il valore del semaforo di n;
    n=0, si legge il valore del semaforo.
